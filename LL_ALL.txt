struct node
{
	int data;
	struct node *next;
};

	node* create_new_node()
{
	int a;
	printf("Enter the value");
	scanf("%d",&a);
	node *p;
	p=(node*)malloc(sizeof(node));
	p->data=a;
	p->next='\0';
	return p;
}
